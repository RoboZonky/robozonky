42  class SimpleInvestmentStrategy implements InvestmentStrategy {
43  
44      private static final Logger LOGGER = LoggerFactory.getLogger(SimpleInvestmentStrategy.class);
45      private static final Comparator<Loan> BY_TERM =
46              (l1, l2) -> Integer.compare(l1.getTermInMonths(), l2.getTermInMonths());
47  
48      static Map<Rating, Collection<Loan>> sortLoansByRating(final Collection<Loan> loans) {
49          final Map<Rating, Collection<Loan>> result = new EnumMap<>(Rating.class);
50          loans.forEach(l -> result.compute(l.getRating(), (k, v) -> {
51              final Collection<Loan> values = (v == null) ? new LinkedHashSet<>(loans.size() / 2) : v;
52              values.add(l);
53              return values;
54          }));
55          return Collections.unmodifiableMap(result);
56      }
57  
58      private List<Rating> rankRatingsByDemand(final Map<Rating, BigDecimal> currentShare,
59                                               final Function<StrategyPerRating, BigDecimal> metric) {
60          final SortedMap<BigDecimal, List<Rating>> mostWantedRatings = new TreeMap<>(Comparator.reverseOrder());
61          // put the ratings into buckets based on how much we're missing them
62          currentShare.forEach((r, currentRatingShare) -> {
63              final BigDecimal maximumAllowedShare = metric.apply(this.individualStrategies.get(r));
64              final BigDecimal undershare = maximumAllowedShare.subtract(currentRatingShare);
65              if (undershare.compareTo(BigDecimal.ZERO) <= 0) { // we over-invested into this rating; do not include
66                  return;
67              }
68              mostWantedRatings.compute(undershare, (k, v) -> { // each rating unique at source, so list works
69                  final List<Rating> target = (v == null) ? new ArrayList<>(1) : v;
70                  target.add(r);
71                  return target;
72              });
73          });
74          return mostWantedRatings.entrySet().stream()
75                  .map(Map.Entry::getValue)
76                  .reduce(new ArrayList<>(Rating.values().length), (a, b) -> {
77                      a.addAll(b);
78                      return a;
79                  });
80      }
81  
82      /**
83       * Ranks the ratings in the portfolio which need to have their overall share increased. First on this list will be
84       * ratings which are under their target share. Following them will be ratings which are under their maximum share.
85       * Ratings with their respective shares over the maximum will not be present.
86       *
87       * @param currentShare Current share of investments in a given rating.
88       * @return Ratings in the order of decreasing demand.
89       */
90      List<Rating> rankRatingsByDemand(final Map<Rating, BigDecimal> currentShare) {
91          final List<Rating> ratingsUnderTarget = rankRatingsByDemand(currentShare, StrategyPerRating::getTargetShare);
92          final List<Rating> ratingsUnderMaximum = rankRatingsByDemand(currentShare, StrategyPerRating::getMaximumShare);
93          Collections.reverse(ratingsUnderMaximum); // the closer we get to maximum share, the less desirable
94          final List<Rating> result = Stream.concat(ratingsUnderTarget.stream(), ratingsUnderMaximum.stream())
95                  .distinct()
96                  .collect(Collectors.toList());
97          return Collections.unmodifiableList(result);
98      }

100     private final int minimumBalance, investmentCeiling;
101     private final Map<Rating, StrategyPerRating> individualStrategies = new EnumMap<>(Rating.class);
102 
103     SimpleInvestmentStrategy(final int minimumBalance, final int investmentCeiling,
104                              final Map<Rating, StrategyPerRating> individualStrategies) {
105         this.minimumBalance = minimumBalance;
106         this.investmentCeiling = investmentCeiling;
107         for (final Rating r: Rating.values()) {
108             if (!individualStrategies.containsKey(r)) {
109                 throw new IllegalArgumentException("Missing strategy for rating " + r);
110             }
111             final StrategyPerRating s = individualStrategies.get(r);
112             this.individualStrategies.put(r, s);
113         }
114     }
115 
116     private boolean isAcceptable(final PortfolioOverview portfolio) {
117         final int availableBalance = portfolio.getCzkAvailable();
118         if (availableBalance < this.minimumBalance) {
119             SimpleInvestmentStrategy.LOGGER.debug("According to the investment strategy, {} CZK balance is less than "
120                     + "minimum {} CZK. Not recommending any loans.", availableBalance, this.minimumBalance);
121             return false;
122         }
123         final int invested = portfolio.getCzkInvested();
124         if (invested > this.investmentCeiling) {
125             SimpleInvestmentStrategy.LOGGER.debug("According to the investment strategy, {} CZK total investment "
126                     + "exceeds {} CZK ceiling. Not recommending any loans.", invested, this.investmentCeiling);
127             return false;
128         }
129         return true;
130     }
131 
132     @Override
133     public List<Loan> getMatchingLoans(final List<Loan> availableLoans, final PortfolioOverview portfolio) {
134         if (!this.isAcceptable(portfolio)) {
135             return Collections.emptyList();
136         }
137         final List<Rating> mostWantedRatings = this.rankRatingsByDemand(portfolio.getSharesOnInvestment());
138         SimpleInvestmentStrategy.LOGGER.info("According to the investment strategy, the portfolio is low "
139                 + "on following ratings: {}. The most under-invested come first.", mostWantedRatings);
140         final Map<Rating, Collection<Loan>> splitByRating = SimpleInvestmentStrategy.sortLoansByRating(availableLoans);
141         final List<Loan> acceptableLoans = new ArrayList<>(availableLoans.size());
142         mostWantedRatings.forEach(rating -> {
143             final Collection<Loan> loans = splitByRating.get(rating);
144             if (loans == null || loans.isEmpty()) { // no loans of this rating
145                 return;
146             }
147             final Comparator<Loan> properOrder = this.individualStrategies.get(rating).isPreferLongerTerms() ?
148                     SimpleInvestmentStrategy.BY_TERM.reversed() : SimpleInvestmentStrategy.BY_TERM;
149             final Collection<Loan> acceptable = loans.stream().sorted(properOrder)
150                     .filter(l -> this.individualStrategies.get(rating).isAcceptable(l))
151                     .collect(Collectors.toList());
152             acceptableLoans.addAll(acceptable);
153         });
154         return Collections.unmodifiableList(acceptableLoans);
155     }
156 
157     @Override
158     public int recommendInvestmentAmount(final Loan loan, final PortfolioOverview portfolio) {
159         if (!this.isAcceptable(portfolio)) {
160             return 0;
161         }
162         final Optional<int[]> recommend = individualStrategies.get(loan.getRating()).recommendInvestmentAmount(loan);
163         if (!recommend.isPresent()) { // does not match the strategy
164             return 0;
165         }
166         final int minimumRecommendation = recommend.get()[0];
167         final int maximumRecommendation = recommend.get()[1];
168         SimpleInvestmentStrategy.LOGGER.debug("Recommended investment range of <{}; {}> CZK.", minimumRecommendation,
169                 maximumRecommendation);
170         // round to nearest lower increment
171         final int balance = portfolio.getCzkAvailable();
172         if (minimumRecommendation > balance) {
173             return 0;
174         } else if (minimumRecommendation > loan.getRemainingInvestment()) {
175             return 0;
176         }
177         final int maxAllowedInvestmentIncrement = InvestmentStrategy.MINIMAL_INVESTMENT_INCREMENT;
178         final int recommendedAmount = Math.min(balance, maximumRecommendation); // to never go over balance
179         final int result = (recommendedAmount / maxAllowedInvestmentIncrement) * maxAllowedInvestmentIncrement;
180         if (result < minimumRecommendation) {
181             return 0;
182         } else {
183             return result;
184         }
185     }
