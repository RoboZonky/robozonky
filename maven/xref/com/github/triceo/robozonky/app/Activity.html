29  import java.util.Comparator;
30  import java.util.List;
31  import java.util.stream.Collectors;

33  import com.github.triceo.robozonky.remote.Api;

38  /**
39   * Decides whether or not the application should fall asleep because of general marketplace inactivity. Uses two sources
40   * of data to make the decision: the marketplace, and the app's internal state concerning the last time the marketplace
41   * was checked.
42   *
43   * In order for the state to be persisted, the App needs to eventually call {@link #settle()} after calling
44   * {@link #shouldSleep()}.
45   */
46  class Activity {
47  
48      /**
49       * Simple abstraction over {@link Api#getLoans()} which provides additional intelligence. Use {@link #from(Api)} as
50       * entry point to the API.
51       */
52      private static class Marketplace {
53  
54          /**
55           * Instantiate the marketplace.
56           *
57           * @param api Remote API from which to load all loans.
58           * @return Marketplace backed by the API.
59           */
60          public static Activity.Marketplace from(final Api api) {
61              return new Activity.Marketplace(api.getLoans());
62          }
63  
64          private final List<Loan> recentLoansDescending;
65  
66          private Marketplace(final Collection<Loan> loans) { // Zotify occasionally returns null loans for unknown reason
67              this.recentLoansDescending = loans == null ? Collections.emptyList() :
68                      Collections.unmodifiableList(loans.stream()
69                              .filter(l -> l.getRemainingInvestment() > 0)
70                              .sorted(Comparator.comparing(Loan::getDatePublished).reversed())
71                              .collect(Collectors.toList()));
72          }
73  
74          /**
75           * Retrieve all loans in the marketplace which have not yet been fully funded and which have been published at least
76           * a certain time ago.
77           * @param delayInSeconds How long ago at the very least should the loans have been published.
78           * @return Ordered by publishing time descending.
79           */
80          public List<Loan> getLoansOlderThan(final int delayInSeconds) {
81              return this.recentLoansDescending.stream()
82                      .filter(l -> Instant.now().isAfter(l.getDatePublished().plus(delayInSeconds, ChronoUnit.SECONDS)))
83                      .collect(Collectors.toList());
84          }
85  
86          /**
87           * Retrieve all loans in the marketplace which have not yet been fully funded and which have been published past a
88           * certain point in time.
89           * @param instant The earliest point in time for the loans to published on.
90           * @return Ordered by publishing time descending.
91           */
92          public List<Loan> getLoansNewerThan(final Instant instant) {
93              return this.recentLoansDescending.stream()
94                      .filter(l -> l.getDatePublished().isAfter(instant))
95                      .collect(Collectors.toList());
96          }
97  
98      }
99  
100     private static final Logger LOGGER = LoggerFactory.getLogger(Activity.class);
101 
102     private final int closedSeasonInSeconds;
103     private final int sleepIntervalInMinutes;
104     private final Path state;
105     private final Activity.Marketplace marketplace;
106     private Runnable settler = null;
107 
108     Activity(final AppContext ctx, final Api api, final Path state) {
109         this.closedSeasonInSeconds = ctx.getCaptchaDelayInSeconds();
110         this.sleepIntervalInMinutes = ctx.getSleepPeriodInMinutes();
111         this.marketplace = Activity.Marketplace.from(api);
112         this.state = state;
113     }
114 
115     private boolean wasMarketplaceCheckedBefore() {
116         return this.state.toFile().exists();
117     }
118 
119     private Instant getLatestMarketplaceAction() {
120         if (!this.wasMarketplaceCheckedBefore()) {
121             return Instant.EPOCH;
122         }
123         try (final BufferedReader reader = Files.newBufferedReader(this.state, StandardCharsets.UTF_8)) {
124             final String instantString = reader.readLine();
125             return Instant.parse(instantString);
126         } catch (final IOException ex) {
127             Activity.LOGGER.debug("Failed read marketplace timestamp.", ex);
128             return Instant.EPOCH;
129         }
130     }
131 
132     private List<Loan> getUnactionableLoans() {
133         return this.marketplace.getLoansNewerThan(Instant.now().minus(this.closedSeasonInSeconds, ChronoUnit.SECONDS));
134     }
135 
136     /**
137      * Retrieves loans that are available for robotic investing, ie. not protected by CAPTCHA.
138      *
139      * @return Loans ordered by their time of publishing, descending.
140      */
141     public List<Loan> getAvailableLoans() {
142         return this.marketplace.getLoansOlderThan(closedSeasonInSeconds);
143     }
144 
145     /**
146      * Whether or not the application should fall asleep and not make any further contact with API.
147      *
148      * @return True if no further contact should be made during this run of the app.
149      */
150     public boolean shouldSleep() {
151         final Instant lastKnownMarketplaceAction = this.getLatestMarketplaceAction();
152         final boolean hasUnactionableLoans = !this.getUnactionableLoans().isEmpty();
153         Activity.LOGGER.debug("Marketplace last checked on {}, has un-actionable loans: {}.",
154                 lastKnownMarketplaceAction, hasUnactionableLoans);
155         boolean shouldSleep = true;
156         if (!this.marketplace.getLoansNewerThan(lastKnownMarketplaceAction).isEmpty()) {
157             // try investing since there are loans we haven't seen yet
158             Activity.LOGGER.debug("Will not sleep due to new loans.");
159             shouldSleep = false;
160         } else if (lastKnownMarketplaceAction.plus(sleepIntervalInMinutes, ChronoUnit.MINUTES).isBefore(Instant.now())) {
161             // try investing since we haven't tried in a while; maybe we have some more funds now
162             Activity.LOGGER.debug("Will not sleep due to already sleeping too much.");
163             shouldSleep = false;
164         }
165         synchronized (this) { // do not allow concurrent modification of the settler variable
166             if (this.settler != null) {
167                 Activity.LOGGER.warn("Scrapping unsettled activity.");
168             }
169             if (!shouldSleep || hasUnactionableLoans) {
170             /*
171              * only persist (= change marketplace check timestamp) when we're intending to execute some actual
172              * investing.
173              */
174                 this.settler = () -> this.persist(hasUnactionableLoans);
175             } else {
176                 this.settler = null;
177             }
178         }
179         return shouldSleep;
180     }
181 
182     /**
183      * Persists the new marketplace state following a {@link #shouldSleep()} call.
184      */
185     public synchronized void settle() {
186         if (this.settler == null) {
187             Activity.LOGGER.debug("No activity to settle.");
188         } else {
189             this.settler.run();
190             this.settler = null;
191         }
192     }
193 
194     private void persist(final boolean hasUnactionableLoans) {
195         try (final BufferedWriter writer = Files.newBufferedWriter(this.state, StandardCharsets.UTF_8)) {
196             // make sure the unactionable loans are never included in the time the marketplace was last checked
197             final Instant result = hasUnactionableLoans ?
198                     Instant.now().minus(this.closedSeasonInSeconds + 30, ChronoUnit.SECONDS)
199                     : Instant.now();
200             if (hasUnactionableLoans) {
201                 Activity.LOGGER.debug("New marketplace last checked time placed before beginning of closed season: {}.",
202                         result);
203             } else {
204                 Activity.LOGGER.debug("New marketplace last checked time is {}.", result);
205             }
206             writer.write(result.toString());
207         } catch (final IOException ex) {
208             Activity.LOGGER.info("Failed write marketplace timestamp, sleep feature will be disabled.", ex);
209             try { // failed to write new data, try to delete stale data
210                 Files.delete(this.state);
211             } catch (final IOException ex2) {
212                 Activity.LOGGER.warn("Failed delete marketplace timestamp, sleep feature may malfunction.", ex2);
213             }
214         }
215     }
