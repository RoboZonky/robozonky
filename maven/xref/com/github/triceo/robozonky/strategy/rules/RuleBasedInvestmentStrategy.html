43  /**
44   * This strategy implements evaluation using a Drools decision table. See http://www.drools.org/
45   */
46  class RuleBasedInvestmentStrategy implements InvestmentStrategy {
47  
48      private static final Logger LOGGER = LoggerFactory.getLogger(RuleBasedInvestmentStrategy.class);
49  
50      private static LoanDescriptor matchLoan(final AcceptedLoan l, final Collection<LoanDescriptor> loans) {
51          return loans.stream()
52                  .filter(ld -> ld.getLoan().getId() == l.getId())
53                  .findFirst()
54                  .orElseThrow(() -> new IllegalStateException("Could not find matching loan. Technically impossible."));
55      }
56  
57      private final KieContainer kieContainer;
58  
59      RuleBasedInvestmentStrategy(final KieContainer kieContainer) {
60          this.kieContainer = kieContainer;
61      }
62  
63      @Override
64      @SuppressWarnings("unchecked")
65      public List<Recommendation> recommend(final Collection<LoanDescriptor> availableLoans,
66                                            final PortfolioOverview portfolio) {
67          RuleBasedInvestmentStrategy.LOGGER.trace("Started matching loans.");
68          final KieSession session = this.kieContainer.newKieSession();
69          // insert facts into session
70          availableLoans.stream().map(ProposedLoan::new).forEach(session::insert);
71          Arrays.stream(Rating.values())
72                  .map(r -> new RatingShare(r, portfolio.getShareOnInvestment(r)))
73                  .forEach(session::insert);
74          session.insert(new Wallet(portfolio.getCzkAvailable(), portfolio.getCzkInvested()));
75          // reason over facts and process results
76          RuleBasedInvestmentStrategy.LOGGER.trace("Facts inserted into session.");
77          session.fireAllRules();
78          RuleBasedInvestmentStrategy.LOGGER.trace("Drools finished.");
79          final Collection<AcceptedLoan> result =
80                  (Collection<AcceptedLoan>)session.getObjects(o -> o instanceof AcceptedLoan);
81          session.dispose();
82          RuleBasedInvestmentStrategy.LOGGER.trace("Session disposed.");
83          final Map<AcceptedLoan, LoanDescriptor> map = result.stream()
84                  .collect(Collectors.toMap(Function.identity(),
85                          l -> RuleBasedInvestmentStrategy.matchLoan(l, availableLoans)));
86          final List<Recommendation> recommendations = map.entrySet().stream()
87                  .sorted(Comparator.comparing(Map.Entry::getKey))
88                  .map(e -> e.getValue().recommend(e.getKey().getAmount(), e.getKey().isConfirmationRequired()))
89                  .filter(Optional::isPresent)
90                  .map(Optional::get)
91                  .collect(Collectors.toList());
92          RuleBasedInvestmentStrategy.LOGGER.trace("Loans matched.");
93          return Collections.unmodifiableList(recommendations);
94      }
