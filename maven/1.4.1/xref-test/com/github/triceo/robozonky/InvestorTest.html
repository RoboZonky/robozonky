61      @Test
62      public void mergingTwoInvestmentCollectionsWorksProperly() {
63          final Investment I1 = InvestorTest.getMockInvestmentWithId(1);
64          final Investment I2 = InvestorTest.getMockInvestmentWithId(2);
65          final Investment I3 = InvestorTest.getMockInvestmentWithId(3);
66  
67          // two identical investments will result in one
68          final List<Investment> a = Arrays.asList(I1, I2);
69          final List<Investment> b = Arrays.asList(I2, I3);
70          Assertions.assertThat(Investor.mergeInvestments(a, b)).containsExactly(I1, I2, I3);
71  
72          // toy around with empty lists
73          Assertions.assertThat(Investor.mergeInvestments(Collections.emptyList(), Collections.emptyList())).isEmpty();
74          Assertions.assertThat(Investor.mergeInvestments(Collections.emptyList(), a))
75                  .containsExactly((Investment[]) a.toArray());
76          Assertions.assertThat(Investor.mergeInvestments(b, Collections.emptyList()))
77                  .containsExactly((Investment[]) b.toArray());
78  
79          // standard merging also works
80          final List<Investment> c = Collections.singletonList(I3);
81          Assertions.assertThat(Investor.mergeInvestments(a, c)).containsExactly(I1, I2, I3);
82  
83          // reverse-order merging works
84          final List<Investment> d = Arrays.asList(I2, I1);
85          Assertions.assertThat(Investor.mergeInvestments(a, d)).containsExactly(I1, I2);
86  
87          // two non-identical loans with same ID are merged in the order in which they came
88          final Investment I3_2 = InvestorTest.getMockInvestmentWithId(3);
89          final List<Investment> e = Collections.singletonList(I3_2);
90          Assertions.assertThat(Investor.mergeInvestments(c, e)).containsExactly(I3);
91          Assertions.assertThat(Investor.mergeInvestments(e, c)).containsExactly(I3_2);
92      }

94      @Test
95      public void recoverOnFailedInvestment() {
96          // the strategy will recommend two different investments
97          final Loan mockLoan1 = InvestorTest.getMockLoanWithId(1);
98          final Loan mockLoan2 = InvestorTest.getMockLoanWithId(2);
99          final InvestmentStrategy strategyMock = Mockito.mock(InvestmentStrategy.class);
100         Mockito.when(strategyMock.getMatchingLoans(Matchers.any(), Matchers.any()))
101                 .thenReturn(Arrays.asList(mockLoan1, mockLoan2));
102         Mockito.when(strategyMock.recommendInvestmentAmount(Matchers.any(), Matchers.any())).thenReturn(400);
103         // fail on the first loan, accept the second
104         final InvestingZonkyApi mockApi = Mockito.mock(InvestingZonkyApi.class);
105         Mockito.doThrow(BadRequestException.class)
106                 .when(mockApi).invest(Matchers.argThat(new InvestorTest.InvestmentBaseMatcher(mockLoan1)));
107         // finally test
108         final Investor investor = new Investor(mockApi, Mockito.mock(ZotifyApi.class), strategyMock,
109                 BigDecimal.valueOf(1000));
110         final Optional<Investment> result = investor.investOnce(BigDecimal.valueOf(1000), new Statistics(),
111                 Collections.emptyList());
112         // check that the first loan properly failed over to the second
113         Mockito.verify(mockApi, Mockito.times(2)).invest(Matchers.any());
114         final SoftAssertions softly = new SoftAssertions();
115         softly.assertThat(result).isPresent();
116         softly.assertThat(result.get().getLoanId()).isEqualTo(mockLoan2.getId());
117         softly.assertAll();
118     }

120     @Test
121     public void investOnRecommendations() {
122         // prepare loans so that only one will survive
123         final BigDecimal balance = BigDecimal.valueOf(1000);
124         final Loan overBalance = InvestorTest.getMockLoanWithIdAndAmount(1, balance.intValue() * 2);
125         final Loan underMinimum =
126                 InvestorTest.getMockLoanWithIdAndAmount(2, InvestmentStrategy.MINIMAL_INVESTMENT_ALLOWED);
127         final int amount = 500;
128         final Loan overAmount = InvestorTest.getMockLoanWithIdAndAmount(3, amount);
129         final Loan success = InvestorTest.getMockLoanWithIdAndAmount(5, amount);
130         final Statistics stats = Mockito.mock(Statistics.class);
131         Mockito.when(stats.getRiskPortfolio()).thenReturn(Collections.emptyList());
132         // prepare pre-conditions for the above loans
133         final InvestmentStrategy strategy = Mockito.mock(InvestmentStrategy.class);
134         Mockito.when(strategy.recommendInvestmentAmount(Matchers.eq(overBalance), Matchers.any()))
135                 .thenReturn(balance.intValue() + 1);
136         Mockito.when(strategy.recommendInvestmentAmount(Matchers.eq(underMinimum), Matchers.any())).thenReturn(0);
137         Mockito.when(strategy.recommendInvestmentAmount(Matchers.eq(overAmount), Matchers.any()))
138                 .thenReturn(amount + 1);
139         Mockito.when(strategy.recommendInvestmentAmount(Matchers.eq(success), Matchers.any())).thenReturn(amount / 2);
140         final InvestingZonkyApi api = Mockito.mock(InvestingZonkyApi.class);
141         final ZotifyApi zotifyApi = Mockito.mock(ZotifyApi.class);
142         // and now actually test that the succeeding loan will be invested into ...
143         final Investor i = new Investor(api, zotifyApi, strategy, balance);
144         Mockito.when(strategy.getMatchingLoans(Matchers.any(), Matchers.any()))
145                 .thenReturn(Arrays.asList(overBalance, underMinimum, overAmount, success));
146         final Optional<Investment> result = i.investOnce(balance, stats, Collections.emptyList());
147         Assertions.assertThat(result).isPresent();
148         Assertions.assertThat(result.get().getLoanId()).isEqualTo(success.getId());
149         Mockito.verify(api, Mockito.times(1)).invest(Matchers.any());
150         // ... no matter which place it takes
151         Mockito.when(strategy.getMatchingLoans(Matchers.any(), Matchers.any()))
152                 .thenReturn(Arrays.asList(success, overBalance, underMinimum, overAmount));
153         final Optional<Investment> result2 = i.investOnce(balance, stats, Collections.emptyList());
154         Assertions.assertThat(result2).isPresent();
155         Assertions.assertThat(result2.get().getLoanId()).isEqualTo(success.getId());
156         Mockito.verify(api, Mockito.times(2)).invest(Matchers.any());
157         // ... even when nothing is accepted
158         final Loan alreadyPresent = InvestorTest.getMockLoanWithIdAndAmount(6, 10000);
159         Mockito.when(strategy.getMatchingLoans(Matchers.any(), Matchers.any()))
160                 .thenReturn(Arrays.asList(overBalance, underMinimum, overAmount, alreadyPresent));
161         final Investment alreadyPresentInvestment = new Investment(alreadyPresent, 200);
162         final Optional<Investment> result3 =
163                 i.investOnce(balance, stats, Collections.singletonList(alreadyPresentInvestment));
164         Assertions.assertThat(result3).isEmpty();
165     }

167     private static class InvestmentBaseMatcher extends BaseMatcher<Investment> {
168         private final Loan matching;
169 
170         public InvestmentBaseMatcher(final Loan matching) {
171             this.matching = matching;
172         }
173 
174         @Override
175         public void describeTo(final Description description) {
176             description.appendText("Matches only the investment matching a specified loan.");
177         }
178 
179         @Override
180         public boolean matches(final Object item) {
181             return ((Investment) item).getLoanId() == matching.getId();
182         }
183     }

185     @Test
186     public void properBlockedAmountRetrieval() {
187         final int loan1id = 1, loan1amount = 100;
188         final Loan l1 = InvestorTest.getMockLoanWithIdAndAmount(loan1id, loan1amount);
189         final int loan2Id = 2, loan2amount = 200;
190         final Loan l2 = InvestorTest.getMockLoanWithIdAndAmount(loan2Id, loan2amount);
191         final ZonkyApi api = Mockito.mock(ZonkyApi.class);
192         Mockito.when(api.getLoan(Matchers.eq(loan1id))).thenReturn(l1);
193         Mockito.when(api.getLoan(Matchers.eq(loan2Id))).thenReturn(l2);
194         Mockito.when(api.getBlockedAmounts(Matchers.anyInt(), Matchers.anyInt())).thenReturn(
195                 Arrays.asList(new BlockedAmount(0, 1000), new BlockedAmount(loan1id, loan1amount),
196                         new BlockedAmount(loan2Id, loan2amount))
197         );
198         final List<Investment> result = Investor.retrieveInvestmentsRepresentedByBlockedAmounts(api);
199         // the 0 ID blocked amount is Zonky's investors' fee, which should not be looked up as a loan
200         Assertions.assertThat(result).hasSize(2);
201         Assertions.assertThat(result.get(0).getLoanId()).isEqualTo(l1.getId());
202         Assertions.assertThat(result.get(1).getLoanId()).isEqualTo(l2.getId());
203     }
