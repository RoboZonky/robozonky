17  package com.github.triceo.robozonky.app;
18  
19  import java.io.File;
20  import java.math.BigDecimal;
21  import java.util.function.BiFunction;
22  import java.util.function.Function;
23  
24  import com.github.triceo.robozonky.Operations;
25  import com.github.triceo.robozonky.remote.Rating;
26  import com.github.triceo.robozonky.strategy.InvestmentStrategy;
27  import com.github.triceo.robozonky.strategy.StrategyBuilder;
28  import org.apache.commons.configuration2.ImmutableConfiguration;
29  import org.apache.commons.configuration2.PropertiesConfiguration;
30  import org.apache.commons.configuration2.builder.FileBasedConfigurationBuilder;
31  import org.apache.commons.configuration2.builder.fluent.Parameters;
32  import org.apache.commons.configuration2.builder.fluent.PropertiesBuilderParameters;
33  import org.apache.commons.configuration2.ex.ConfigurationException;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  class StrategyParser {
38  
39      private static final Logger LOGGER = LoggerFactory.getLogger(StrategyParser.class);
40  
41      private static <T> T getValue(final ImmutableConfiguration config, final Rating r, final String property, final Function<String, T> supplier) {
42          final String propertyName = property + '.' + r.name();
43          final String fallbackPropertyName = property + ".default";
44          if (config.containsKey(propertyName)) {
45              return supplier.apply(propertyName);
46          } else if (config.containsKey(fallbackPropertyName)) {
47              return supplier.apply(fallbackPropertyName);
48          } else {
49              throw new IllegalStateException("Investment strategy is incomplete. Missing value for '" + property + "' and rating '" + r + '\'');
50          }
51      }
52  
53      private static <T> T getValue(final ImmutableConfiguration config, final Rating r, final String property, final BiFunction<String, T, T> supplier, final T def) {
54          return StrategyParser.getValue(config, r, property, (a) -> supplier.apply(a, def));
55      }
56  
57      private static ImmutableConfiguration getConfig(final File strategyFile) throws ConfigurationException {
58          // read config file
59          final PropertiesBuilderParameters props = new Parameters().properties().setFile(strategyFile);
60          final FileBasedConfigurationBuilder<PropertiesConfiguration> builder = new FileBasedConfigurationBuilder<>(PropertiesConfiguration.class);
61          builder.configure(props);
62          return builder.getConfiguration();
63      }
64  
65      public static InvestmentStrategy parse(final File strategyFile) throws ConfigurationException {
66          StrategyParser.LOGGER.debug("Will parse strategy file: '{}'", strategyFile.getAbsolutePath());
67          final ImmutableConfiguration config = StrategyParser.getConfig(strategyFile);
68          final StrategyBuilder strategies = new StrategyBuilder();
69          BigDecimal sumShares = BigDecimal.ZERO;
70          for (final Rating rating : Rating.values()) { // prepare strategy for a given rating
71              final boolean preferLongerTerms =
72                      StrategyParser.getValue(config, rating, "preferLongerTerms", config::getBoolean);
73              final BigDecimal targetShare =
74                      StrategyParser.getValue(config, rating, "targetShare", config::getBigDecimal);
75              if (targetShare.compareTo(BigDecimal.ZERO) < 0 || targetShare.compareTo(BigDecimal.ONE) > 0) {
76                  throw new IllegalStateException("Target share for rating " + rating + " outside of range <0, 1>: "
77                          + targetShare);
78              }
79              sumShares = sumShares.add(targetShare);
80              final int minTerm = StrategyParser.getValue(config, rating, "minimumTerm", config::getInteger, 0);
81              if (minTerm < -1) {
82                  throw new IllegalStateException("Minimum acceptable term for rating " + rating + " is negative.");
83              }
84              final int maxTerm = StrategyParser.getValue(config, rating, "maximumTerm", config::getInteger, 0);
85              if (maxTerm < -1) {
86                  throw new IllegalStateException("Maximum acceptable term for rating " + rating + " is negative.");
87              } else if (minTerm > maxTerm && maxTerm != -1) {
88                  throw new IllegalStateException("Maximum acceptable term for rating " + rating
89                          + " is smaller than the minimum.");
90              }
91              final int maxLoanAmount =
92                      StrategyParser.getValue(config, rating, "maximumLoanAmount", config::getInteger, 0);
93              if (maxLoanAmount < Operations.MINIMAL_INVESTMENT_ALLOWED) {
94                  throw new IllegalStateException("Maximum investment amount for rating " + rating
95                          + "  is smaller than minimum.");
96              }
97              final BigDecimal maxLoanShare =
98                      StrategyParser.getValue(config, rating, "maximumLoanShare", config::getBigDecimal);
99              if (maxLoanShare.compareTo(BigDecimal.ZERO) < 0 || maxLoanShare.compareTo(BigDecimal.ONE) > 0) {
100                 throw new IllegalStateException("Maximum investment share for rating " + rating
101                         + " outside of range <0, 1>: " + targetShare);
102             }
103             strategies.addIndividualStrategy(rating, targetShare, minTerm, maxTerm, maxLoanAmount, maxLoanShare,
104                     preferLongerTerms);
105         }
106         if (sumShares.compareTo(BigDecimal.ONE) > 0) {
107             StrategyParser.LOGGER.warn("Sum of target shares ({}) is larger than 1. Some ratings are likely to be " +
108                     "over-represented.", sumShares);
109         } else if (sumShares.compareTo(BigDecimal.ONE) < 0) {
110             StrategyParser.LOGGER.warn("Sum of target shares ({}) is smaller than 1. You are likely to leave money " +
111                     "unspent in your wallet.", sumShares);
112         }
113         return strategies.build();
114     }
