46      /**
47       * Determine whether or not a given loan is present among existing investments.
48       *
49       * @param loan Loan in question.
50       * @param investments Known investments.
51       * @return True if present.
52       */
53      private static boolean isLoanPresent(final Loan loan, final Collection<Investment> investments) {
54          return investments.stream().filter(i -> loan.getId() == i.getLoanId()).findFirst().isPresent();
55      }
56  
57      static Collection<Investment> mergeInvestments(final Collection<Investment> left,
58                                                     final Collection<Investment> right) {
59          if (left.isEmpty() && right.isEmpty()) {
60              return Collections.emptyList();
61          } else if (left.isEmpty()) {
62              return Collections.unmodifiableCollection(right);
63          } else if (right.isEmpty()) {
64              return Collections.unmodifiableCollection(left);
65          } else {
66              final Map<Integer, Investment> investments
67                      = left.stream().collect(Collectors.toMap(Investment::getLoanId, Function.identity()));
68              right.stream().filter(investment -> !investments.containsKey(investment.getLoanId()))
69                      .forEach(investment -> investments.put(investment.getLoanId(), investment));
70              return Collections.unmodifiableCollection(investments.values());
71          }
72      }

74      /**
75       * The core investing call. Receives a particular loan, checks if the user has enough money to invest, and sends the
76       * command to the Zonky API.
77       *
78       * @param api Authenticated Zonky API, ready for {@link InvestOperation}.
79       * @param l Loan to invest into.
80       * @param amount Amount to invest into the loan.
81       * @param balance How much usable cash the user has in the wallet.
82       * @return Present if input valid and operation succeeded, empty otherwise.
83       */
84      private static Optional<Investment> invest(final ZonkyApi api, final Loan l, final int amount, final int balance) {
85          if (amount < InvestmentStrategy.MINIMAL_INVESTMENT_ALLOWED) {
86              Investor.LOGGER.info("Not investing into loan '{}', since investment ({} CZK) less than bare minimum.",
87                      l, amount);
88              return Optional.empty();
89          } else if (amount > balance) {
90              Investor.LOGGER.info("Not investing into loan '{}', {} CZK to invest is more than {} CZK balance.",
91                      l, amount, balance);
92              return Optional.empty();
93          } else if (amount > l.getAmount()) {
94              Investor.LOGGER.info("Not investing into loan '{}', {} CZK to invest is more than {} CZK loan amount.",
95                      l, amount, l.getAmount());
96              return Optional.empty();
97          }
98          final Investment investment = new Investment(l, amount);
99          return new InvestOperation().apply(api, investment);
100     }

102     /**
103      * Blocked amounts represent loans in various stages. Either the user has invested and the loan has not yet been
104      * funded to 100 % ("na tržišti"), or the user invested and the loan has been funded ("na cestě"). In the latter
105      * case, the loan has already disappeared from the marketplace, which means that it will not be available for
106      * investing any more. As far as I know, the next stage is "v pořádku", the blocked amount is cleared and the loan
107      * becomes an active investment.
108      *
109      * Based on that, this method deals with the first case - when the loan is still available for investing, but we've
110      * already invested as evidenced by the blocked amount. It also unnecessarily deals with the second case, since
111      * that is represented by a blocked amount as well. But that does no harm.
112      *
113      * @param api Authenticated API that will be used to retrieve the user's blocked amounts from the wallet.
114      * @return Every blocked amount represents a future investment. This method returns such investments.
115      */
116     static List<Investment> retrieveInvestmentsRepresentedByBlockedAmounts(final ZonkyApi api) {
117         return Collections.unmodifiableList(api.getBlockedAmounts().stream()
118                 .filter(blocked -> blocked.getLoanId() > 0) // 0 == Zonky investors' fee
119                 .map(blocked -> {
120                     final int loanId = blocked.getLoanId();
121                     final int loanAmount = blocked.getAmount();
122                     final Investment i = new Investment(api.getLoan(loanId), loanAmount);
123                     Investor.LOGGER.debug("{} CZK is being blocked by loan {}.", loanAmount, loanId);
124                     return i;
125                 }).collect(Collectors.toList()));
126     }

128     private final ZonkyApi zonkyApi;
129     private final ZotifyApi zotifyApi;
130     private final BigDecimal initialBalance;
131     private final InvestmentStrategy strategy;

133     /**
134      * Standard constructor.
135      *
136      * @param zonky Authenticated API ready to retrieve user information.
137      * @param zotify Marketplace cache for reading loans out of.
138      * @param strategy Strategy used to determine the loans to invest in and the amounts to invest into them.
139      * @param initialBalance How much available cash the user has in their wallet.
140      */
141     public Investor(final ZonkyApi zonky, final ZotifyApi zotify, final InvestmentStrategy strategy,
142                     final BigDecimal initialBalance) {
143         this.zonkyApi = zonky;
144         this.zotifyApi = zotify;
145         this.initialBalance = initialBalance;
146         Investor.LOGGER.info("RoboZonky starting account balance is {} CZK.", this.initialBalance);
147         this.strategy = strategy;
148     }
149 
150     /**
151      * Prepares a list of loans that are suitable for investment by asking the strategy. Then goes over that list one
152      * by one, in the order prescribed by the strategy, and attempts to invest into these loans. The first such
153      * investment operation that succeeds will return.
154      *
155      * @param balance How much money the user has in the wallet that can be used for investing.
156      * @param stats User's portfolio coming from the Zonky API.
157      * @param investmentsAlreadyMade Loans already invested into that have not yet disappeared from marketplace.
158      * @return The first {@link #invest(ZonkyApi, Loan, int, int)} which succeeds, or empty if none have.
159      */
160     Optional<Investment> investOnce(final BigDecimal balance, final Statistics stats,
161                                     final Collection<Investment> investmentsAlreadyMade) {
162         final PortfolioOverview portfolio = PortfolioOverview.calculate(balance, stats, investmentsAlreadyMade);
163         Investor.LOGGER.debug("Current share of unpaid loans with a given rating is: {}.",
164                 portfolio.getSharesOnInvestment());
165         return this.strategy.getMatchingLoans(this.zotifyApi.getLoans(), portfolio).stream()
166                 .filter(l -> !Investor.isLoanPresent(l, investmentsAlreadyMade))
167                 .map(l -> {
168                     final int invest = this.strategy.recommendInvestmentAmount(l, portfolio);
169                     return Investor.invest(this.zonkyApi, l, invest, portfolio.getCzkAvailable());
170                 })
171                 .flatMap(o -> o.isPresent() ? Stream.of(o.get()) : Stream.empty())
172                 .findFirst();
173     }

175     /**
176      * One of the two entry points to the investment API. This takes the strategy, determines suitable loans, and tries
177      * to invest in as many of them as the balance allows.
178      *
179      * @return Investments made in the session.
180      */
181     public Collection<Investment> invest() {
182         // make sure we have enough money to invest
183         final BigDecimal minimumInvestmentAmount = BigDecimal.valueOf(InvestmentStrategy.MINIMAL_INVESTMENT_ALLOWED);
184         BigDecimal balance = this.initialBalance;
185         if (balance.compareTo(minimumInvestmentAmount) < 0) {
186             return Collections.emptyList(); // no need to do anything else
187         }
188         Collection<Investment> investments = Investor.retrieveInvestmentsRepresentedByBlockedAmounts(this.zonkyApi);
189         Investor.LOGGER.debug("The following loans are coming from the API as already invested into: {}", investments);
190         final Statistics stats = this.zonkyApi.getStatistics();
191         // and start investing
192         final Collection<Investment> investmentsMade = new ArrayList<>();
193         do {
194             final Optional<Investment> investment = this.investOnce(balance, stats, investments);
195             if (!investment.isPresent()) { // there is nothing to invest into; RoboZonky is finished now
196                 break;
197             }
198             final Investment i = investment.get();
199             investmentsMade.add(i);
200             investments = Investor.mergeInvestments(investments, Collections.singletonList(i));
201             balance = balance.subtract(BigDecimal.valueOf(i.getAmount()));
202             Investor.LOGGER.info("New account balance is {} CZK.", balance);
203         } while (balance.compareTo(minimumInvestmentAmount) >= 0);
204         return Collections.unmodifiableCollection(investmentsMade);
205     }

207     /**
208      * One of the two entry points to the investment API. Takes a particular loan and invests a given amount into it.
209      *
210      * @param loanId ID of the loan that should be invested into.
211      * @param loanAmount Amount in CZK to be invested.
212      * @return Present if investment succeeded, empty otherwise.
213      */
214     public Optional<Investment> invest(final int loanId, final int loanAmount) {
215         return Investor.invest(this.zonkyApi, this.zonkyApi.getLoan(loanId), loanAmount,
216                 this.initialBalance.intValue());
217     }
