17  package com.github.triceo.robozonky;
18  
19  import java.math.BigDecimal;
20  import java.math.RoundingMode;
21  import java.util.Arrays;
22  import java.util.Collections;
23  import java.util.List;
24  
25  import com.github.triceo.robozonky.remote.Investment;
26  import com.github.triceo.robozonky.remote.Rating;
27  import com.github.triceo.robozonky.remote.RiskPortfolio;
28  import com.github.triceo.robozonky.remote.Statistics;
29  import org.assertj.core.api.Assertions;
30  import org.junit.Test;
31  import org.mockito.Mockito;
32  
33  public class PortfolioOverviewTest {
34  
35      private static void assertProperRatingShare(final PortfolioOverview result, final Rating r, final int amount,
36                                                  final int total) {
37          final BigDecimal expectedShare =
38                  BigDecimal.valueOf(amount).divide(BigDecimal.valueOf(total), 4, RoundingMode.HALF_EVEN);
39          Assertions.assertThat(result.getShareOnInvestment(r)).isEqualTo(expectedShare);
40      }
41  
42      private static List<Investment> getMockInvestmentWithBalance(final int loanAmount) {
43          final Investment i = Mockito.mock(Investment.class);
44          Mockito.when(i.getAmount()).thenReturn(loanAmount);
45          return Collections.singletonList(i);
46      }
47  
48      @Test
49      public void properRatingShareCalculation() {
50          // mock necessary structures
51          final int amountAA = 300, amountB = 200, amountD = 100;
52          final int totalPie = amountAA + amountB + amountD;
53          final RiskPortfolio riskAA = new RiskPortfolio(Rating.AA, -1, amountAA, -1, -1);
54          final RiskPortfolio riskB = new RiskPortfolio(Rating.B, -1, amountB, -1, -1);
55          final RiskPortfolio riskD = new RiskPortfolio(Rating.D, -1, amountD, -1, -1);
56          final Statistics stats = Mockito.mock(Statistics.class);
57          Mockito.when(stats.getRiskPortfolio()).thenReturn(Arrays.asList(riskAA, riskB, riskD));
58  
59          // check standard operation
60          final BigDecimal balance = BigDecimal.TEN;
61          PortfolioOverview result = PortfolioOverview.calculate(balance, stats, Collections.emptyList());
62          Assertions.assertThat(result.getCzkAvailable()).isEqualTo(balance.intValue());
63          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AA, amountAA, totalPie);
64          PortfolioOverviewTest.assertProperRatingShare(result, Rating.B, amountB, totalPie);
65          PortfolioOverviewTest.assertProperRatingShare(result, Rating.D, amountD, totalPie);
66          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAAA, 0, totalPie); // test other ratings included
67          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAA, 0, totalPie);
68          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAA, 0, totalPie);
69          PortfolioOverviewTest.assertProperRatingShare(result, Rating.A, 0, totalPie);
70          PortfolioOverviewTest.assertProperRatingShare(result, Rating.C, 0, totalPie);
71          // check operation with offline investments
72          final int increment = 200, newTotalPie = totalPie + increment;
73          final List<Investment> investments = PortfolioOverviewTest.getMockInvestmentWithBalance(increment);
74          final Investment i = investments.get(0);
75          Mockito.when(i.getRating()).thenReturn(Rating.A);
76          result = PortfolioOverview.calculate(balance, stats, investments);
77          Assertions.assertThat(result.getCzkAvailable()).isEqualTo(balance.intValue());
78          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AA, amountAA, newTotalPie);
79          PortfolioOverviewTest.assertProperRatingShare(result, Rating.B, amountB, newTotalPie);
80          PortfolioOverviewTest.assertProperRatingShare(result, Rating.D, amountD, newTotalPie);
81          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAAA, 0, newTotalPie); // test other ratings included
82          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAAA, 0, newTotalPie);
83          PortfolioOverviewTest.assertProperRatingShare(result, Rating.AAA, 0, newTotalPie);
84          PortfolioOverviewTest.assertProperRatingShare(result, Rating.A, increment, newTotalPie);
85          PortfolioOverviewTest.assertProperRatingShare(result, Rating.C, 0, newTotalPie);
86      }
