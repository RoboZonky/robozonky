17  package com.github.triceo.robozonky.util;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.io.OutputStream;
23  import java.nio.file.Files;
24  import java.util.Arrays;
25  import java.util.StringJoiner;
26  
27  public class IoTestUtil {
28  
29      public static File streamToFile(final InputStream s) throws IOException {
30          return IoTestUtil.streamToFile(s, ".tmp");
31      }
32  
33      public static File streamToFile(final InputStream fis, final String extension) throws IOException {
34          final File f = File.createTempFile("robozonky-", extension);
35          // binary copy since the stream is not guaranteed textual
36          try (final OutputStream fos = Files.newOutputStream(f.toPath())) {
37              final byte[] buffer = new byte[1024];
38              int noOfBytes = 0;
39              while ((noOfBytes = fis.read(buffer)) != -1) {
40                  fos.write(buffer, 0, noOfBytes);
41              }
42          }
43          return f;
44      }
45  
46      private static final String ROOT = "src/main/";
47  
48      public static final String findMainSource() {
49          final File f = new File(IoTestUtil.ROOT);
50          if (f.exists()) {
51              return f.getAbsolutePath();
52          }
53          final File f2 = new File("../", IoTestUtil.ROOT); // in case we're running from /target
54          if (f2.exists()) {
55              return f2.getAbsolutePath();
56          }
57          throw new IllegalStateException("Oops.");
58      }
59  
60      public static final String findMainSource(String... subfolders) {
61          final StringJoiner sj = new StringJoiner(File.separator);
62          sj.add(IoTestUtil.findMainSource());
63          Arrays.stream(subfolders).forEach(sj::add);
64          return sj.toString();
65      }
